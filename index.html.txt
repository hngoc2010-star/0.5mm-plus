<script>

const WIDTH=1.22
const LENGTH=2.44

document.querySelectorAll("input").forEach(i=>{
i.addEventListener("input",calc)
})

function v(id){
return parseFloat(document.getElementById(id).value)||0
}

function f(n){
return Number(n).toLocaleString("en-US",{maximumFractionDigits:2})
}

function calc(){

let thickness=v("thickness")
let bundles=v("bundles")
let pcsBundle=v("pcsBundle")

let usd=v("usdRate")

let exwPcs=v("exwPcs")
let exwCbm=v("exwCbm")

let logistics=v("logistics")
let cfr=v("cfr")

let totalPcs=bundles*pcsBundle

let cbmPcs=(thickness/1000)*WIDTH*LENGTH
let cbmSellPcs=((thickness+0.5)/1000)*WIDTH*LENGTH

let cbmBuy=totalPcs*cbmPcs
let cbmSell=totalPcs*cbmSellPcs
let cbmDiff=cbmSell-cbmBuy

let exwTotal=0

if(exwPcs>0){
exwTotal=exwPcs*totalPcs
}
else if(exwCbm>0){
exwTotal=exwCbm*cbmBuy
}

let totalCfr=Math.round(exwTotal+logistics)

let selling=cbmSell*cfr*usd

let profit=Math.round(selling-totalCfr)

let profitCbm=cbmSell>0?Math.round(profit/cbmSell):0

document.getElementById("pcs").innerText=f(totalPcs)

document.getElementById("cbmBuy").innerText=f(cbmBuy)
document.getElementById("cbmSell").innerText=f(cbmSell)
document.getElementById("cbmDiff").innerText=f(cbmDiff)

document.getElementById("totalCfr").innerText=f(totalCfr)

document.getElementById("profitCbm").innerText=f(profitCbm)

let p=document.getElementById("profit")

p.innerText=f(profit)

p.className=profit>=0?"blue":"red"

}

</script>
